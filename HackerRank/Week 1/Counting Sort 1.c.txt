#include<stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    int max=0;
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
        if(arr[i]>max){
            max=arr[i];
        }
    }
    
    
  
    // printf("%d",z);
    int res[100];
   int k=0;
    for(int i=0;i<100;i++){
        int c=0;
        for(int j=0;j<n;j++){
            if(i==arr[j]){
                c++;
            }
        }
        res[k]=c;
        k++;
    }
    for(int i=0;i<100;i++){
        printf("%d ",res[i]);
    }
    return 0;
}
