#include<stdio.h>
#include<stdlib.h>
int main(){
    int n;
    scanf("%d",&n);
    int arr[n][n];
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
                scanf("%d",&arr[i][j]);
            }
    }
    
    int sum1 =0;
    int sum2=0;
    for(int i=0;i<n;i++){
    
                sum1=sum1+arr[i][i];
            
    }
    int k=0;
    int l=n-1;
    for(int i=0;i<n;i++){
        sum2=sum2+arr[k][l];
        k++;
        l--;
    }
    
    printf("%d\n",abs(sum1-sum2));
    return 0;
}
