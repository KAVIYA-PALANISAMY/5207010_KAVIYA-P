#include<stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    
    float z=0;
    float neg=0;
    float pos=0;
    for(int i=0;i<n;i++){
        if(arr[i]==0){
            z++;
        }
        else if(arr[i]<0){
            neg++;
        }
        else{
            pos++;
        }
    }
    printf("%0.6f\n%0.6f\n%0.6f",pos/n,neg/n,z/n);
    return 0;
    }
