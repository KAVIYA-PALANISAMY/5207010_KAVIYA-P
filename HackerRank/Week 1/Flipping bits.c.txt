

#include <stdio.h>
#include <math.h>
unsigned int flipBits(unsigned int n) {
    int  array[32];
    int i = 0;

    while (n != 0 && i < 32) {
         array[i] = n % 2;
        n = n / 2;
        i++;
    }

    for (int j = i; j < 32; j++) {
         array[j] = 0;
    }

    for (int k = 0; k < 32; k++) {
         array[k] = ( array[k] == 0) ? 1 : 0;
    }

    double sum = 0;
    for (int k = 0; k < 32; k++) {
        sum +=  array[k] * pow(2, k);
    }

    return (unsigned int)sum;
}

int main() {
    int num;
    scanf("%d", &num);

    for (int i = 0; i < num; i++) {
        unsigned int n;
        scanf("%u", &n);

        unsigned int result = flipBits(n);
        printf("%u\n", result);
    }

    return 0;
}
